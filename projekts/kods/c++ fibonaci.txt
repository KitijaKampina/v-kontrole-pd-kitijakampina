#include <bits/stdc++.h>
#include <iostream>
#define pb push_back
#define ll long long
#define ull unsigned long long
#define pii pair<int, int>
#define fi first
#define se second
#define yes "YES\n"
#define no "NO\n"
#pragma GCC optimize("O3,unroll-loops")
#pragma GCC target("avx2,bmi,bmi2,popcnt,lzcnt")
using namespace std;
int main()
{
    ios_base::sync_with_stdio(0); cin.tie(0); cout.tie(0);
    vector<ull>v(100);
    v[0]=0;
    v[1]=1;
    for(int i=2; i<100; i++)
        v[i]=v[i-1]+v[i-2];
    reverse(v.begin(), v.end());
    for(auto x : v)
        cout << x << ' ';
    cout << '\n';
    return 0;
}
